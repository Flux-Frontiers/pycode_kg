code_kg

code_kg: A tool to build a searchable knowledge graph from Python repositories.

Pure AST extraction → SQLite (authoritative) → LanceDB (semantic index).

Public API

Primary entry point::

from code_kg import CodeKG

kg = CodeKG(repo_root, db_path, lancedb_dir)
stats = kg.build(wipe=True)
result = kg.query("database connection setup")
pack = kg.pack("configuration loading")
pack.save("context.md")

Individual layers::

from code_kg import CodeGraph, GraphStore, SemanticIndex

Result types::

from code_kg import BuildStats, QueryResult, SnippetPack, Snippet

Low-level primitives (v0 contract, locked)::

from code_kg import Node, Edge

277    def query_nodes(
278        self,
279        *,
280        kinds: Sequence[str] | None = None,
281        module: str | None = None,
282    ) -> list[dict]:
283        """
284        Return nodes matching optional filters.
285
286        :param kinds: Restrict to these node kinds (e.g. ``["function", "method"]``).
287        :param module: Restrict to nodes in this module path (exact match).
288        :return: List of node dicts.
289        """
290        clauses: list[str] = []
291        params: list[object] = []
292
293        if kinds:
294            placeholders = ",".join("?" for _ in kinds)
295            clauses.append(f"kind IN ({placeholders})")
296            params.extend(kinds)

298        if module is not None:
299            clauses.append("module_path = ?")
300            params.append(module)
301
302        where = f"WHERE {' AND '.join(clauses)}" if clauses else ""
303        rows = self.con.execute(
304            f"""
305            SELECT id, kind, name, qualname, module_path, lineno, end_lineno, docstring
306            FROM nodes {where}
307            ORDER BY module_path, lineno
308            """,
309            params,
310        ).fetchall()

317    def edges_within(self, node_ids: set[str]) -> list[dict]:
318        """
319        Return all edges where both ``src`` and ``dst`` are in *node_ids*.
320
321        :param node_ids: Set of node IDs to restrict to.
322        :return: List of edge dicts with keys ``src``, ``rel``, ``dst``, ``evidence``.
323        """
324        if not node_ids:
325            return []
326
327        self.con.execute("DROP TABLE IF EXISTS _tmp_ids;")
328        self.con.execute("CREATE TEMP TABLE _tmp_ids (id TEXT PRIMARY KEY);")
329        self.con.executemany("INSERT INTO _tmp_ids (id) VALUES (?)", [(i,) for i in node_ids])
330        rows = self.con.execute(
331            """
332            SELECT e.src, e.rel, e.dst, e.evidence
333            FROM edges e
334            JOIN _tmp_ids s ON s.id = e.src
335            JOIN _tmp_ids d ON d.id = e.dst
336            """
337        ).fetchall()
338        return [{"src": r[0], "rel": r[1], "dst": r[2], "evidence": r[3]} for r in rows]

344    def expand(
345        self,
346        seed_ids: set[str],
347        *,
348        hop: int = 1,
349        rels: tuple[str, ...] = DEFAULT_RELS,
350    ) -> dict[str, ProvMeta]:
351        """
352        Expand the graph from *seed_ids* up to *hop* hops.
353
354        Returns a mapping from every reachable node ID to its
355        :class:`ProvMeta` (minimum hop distance and originating seed).
356
357        :param seed_ids: Starting node IDs (hop 0).
358        :param hop: Maximum number of hops to traverse.
359        :param rels: Edge relation types to follow.
360        :return: ``{node_id: ProvMeta}`` for all reachable nodes.
361        """
362        rels = tuple(rels)
363        meta: dict[str, ProvMeta] = {sid: ProvMeta(best_hop=0, via_seed=sid) for sid in seed_ids}
364        frontier: set[str] = set(seed_ids)
365
366        for h in range(1, hop + 1):
367            nxt: set[str] = set()
368            for nid in frontier:
369                rows = self.con.execute(
370                    f"""
371                    SELECT src, dst FROM edges
372                    WHERE (src = ? OR dst = ?)
373                      AND rel IN ({",".join("?" for _ in rels)})
374                    """,
375                    (nid, nid, *rels),
376                ).fetchall()
377                for src, dst in rows:
378                    for cand in (src, dst):
379                        if cand not in meta:
380                            meta[cand] = ProvMeta(
381                                best_hop=h,
382                                via_seed=meta[nid].via_seed,
383                            )
384                            nxt.add(cand)
385                        elif h < meta[cand].best_hop:
386                            meta[cand] = ProvMeta(
387                                best_hop=h,
388                                via_seed=meta[nid].via_seed,
389                            )
390                            nxt.add(cand)
391            frontier = nxt
392
393        return meta

399    def resolve_symbols(self) -> int:
400        """
401        Add ``RESOLVES_TO`` edges from ``sym:`` stub nodes to their
402        first-party definitions (``fn:``, ``cls:``, ``m:``, ``mod:``).
403
404        The AST visitor emits ``CALLS → sym:Foo`` for any call whose target
405        cannot be resolved at walk time (imported names, attribute accesses,
406        etc.).  This post-build step links those stubs to the canonical
407        definition nodes by matching on the ``name`` field, making fan-in
408        queries (who calls X?) work correctly via graph traversal.

546

549        :param hop: Graph expansion hops.
550        :param rels: Edge types to expand.
551        :param include_symbols: Include ``symbol`` nodes.
552        :param context: Extra context lines around each definition span.
553        :param max_lines: Maximum lines per snippet block (default 60).
554        :param max_nodes: Maximum nodes to return (default 15); ``None`` means no limit.
555        :return: :class:`SnippetPack`.
556        """
557        hits = self.index.search(q, k=k)
558        seed_rank: dict[str, dict] = {h.id: {"rank": h.rank, "dist": h.distance} for h in hits}
559        seed_ids: set[str] = set(seed_rank.keys())
560
561        meta = self.store.expand(seed_ids, hop=hop, rels=rels)
562        all_ids = set(meta.keys())
563
564        # Materialise + annotate nodes
565        raw_nodes: list[dict] = []
566        for nid in sorted(all_ids):
567            n = self.store.node(nid)
568            if not n:
569                continue
570            if not include_symbols and n["kind"] == "symbol":
571                continue
572
573            prov: ProvMeta = meta[nid]
574            base_dist = seed_rank.get(prov.via_seed, {"dist": 1e9})["dist"]
575            kind_pri = _KIND_PRIORITY.get(n["kind"], 99)
576            n["_rank_key"] = (prov.best_hop, base_dist, kind_pri, n["id"])
577            n["_best_hop"] = prov.best_hop
578            n["_via_seed"] = prov.via_seed
579            raw_nodes.append(n)
580
581        # Attach spans (needed for dedup)
582        file_cache: dict[str, list[str]] = {}
583        for n in raw_nodes:
584            mp = n.get("module_path")

595                context=context,

640        for n in kept:

651            model=self.model_name,
652            nodes=kept,
653            edges=edges,
654        )
